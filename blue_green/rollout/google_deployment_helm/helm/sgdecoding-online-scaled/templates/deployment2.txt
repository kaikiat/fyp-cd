apiVersion: argoproj.io/v1alpha1
kind: Rollout
metadata:
  name: {{ include "application.master.name" . }}
  labels:
    app.kubernetes.io/name: {{ include "application.master.name" . }}
    helm.sh/chart: {{ include "application.chart" . }}
    app.kubernetes.io/instance: {{ .Release.Name }}
    app.kubernetes.io/managed-by: {{ .Release.Service }}
spec:
  replicas: 1
  selector:
    matchLabels:
      app.kubernetes.io/name: {{ include "application.master.name" . }}
      app.kubernetes.io/instance: {{ .Release.Name }}
  # Add strategy here
  strategy:
    blueGreen: 
      activeService: {{ include "application.master.name" . }}
      previewService: {{ include "application.master.name" . }}-preview #  to test a new version of an application.
      autoPromotionEnabled: false #  make the rollout automatically promote the new ReplicaSet to the active service once the new ReplicaSet is healthy
  template:
    metadata:
      annotations:
        prometheus.io/scrape: 'true'
        prometheus.io/port: '8081'
      labels:
        app.kubernetes.io/name: {{ include "application.master.name" . }}
        app.kubernetes.io/instance: {{ .Release.Name }}
    spec:
      imagePullSecrets:
        - name: {{ .Values.image.pullSecrets }}
      containers:
        - name: {{ include "application.master.name" . }}
          image: {{ .Values.image.repository }}
          imagePullPolicy: {{ .Values.image.pullPolicy }}
          command: [{{ include "application.master.command" . }}]
          envFrom:
          - secretRef:
              name: environment-variables-master-secret
          ports:
            - name: http
              containerPort: 8080
              protocol: TCP
            - name: prometheus
              containerPort: 8081
          livenessProbe:
            httpGet:
              path: /
              {{- if eq .Values.service.enablehttps true}}
              scheme: HTTPS
              port: 8080
              {{- else }}
              port: http
              {{- end}}
            periodSeconds: 60
          readinessProbe:
            httpGet:
              path: /
              {{- if eq .Values.service.enablehttps true}}
              scheme: HTTPS
              port: 8080
              {{- else }}
              port: http
              {{- end}}
            periodSeconds: 60
---
